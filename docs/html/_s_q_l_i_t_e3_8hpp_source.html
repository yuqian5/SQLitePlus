196  int execute(SQLITE3_QUERY &query) {
197  exec_lock->lock(); // lock exec
198 
199  // check if database connection is open
200  if (!*db) {
201  error_no = UNINITIALIZED_ERROR;
202  exec_lock->unlock(); // unlock exec
203 
204  return 1;
205  }
206 
207  // get query from SQLITE3_QUERY
208  std::string prepared_query;
209  try {
210  prepared_query = query.bind().bound_query;
211  } catch (std::out_of_range &e) {
212  error_no = QUERY_BINDING_ERROR;
213  exec_lock->unlock(); // unlock exec
214 
215  return 1;
216  }
217 
218  // clear result and column vector
219  result->clear();
220  column_name->clear();
221 
222  // run query
223  auto data_pack = Callback_Data(column_name, result);
224  int rc = sqlite3_exec(*db, prepared_query.c_str(), &exec_callback, &data_pack, err_msg.get());
225  if (rc != SQLITE_OK) { // check for error
226  // copy error message or get error message
227  if (*err_msg) {
228  err_msg_str = std::string(*err_msg);
229  sqlite3_free(*err_msg);
230  } else {
231  err_msg_str = std::string(sqlite3_errmsg(*db));
232  }
233 
234  error_no = EXECUTION_ERROR;
235 
236  exec_lock->unlock(); // unlock exec
237 
238  return 1;
239  }
240 
241  exec_lock->unlock(); // unlock exec
242  return 0;
243  }

504  static int exec_callback(void *ptr, int argc, char *argv[], char *col_name[]) {
505  auto *data = reinterpret_cast<Callback_Data *>(ptr);
506 
507  // record column name if needed
508  if (data->col->empty()) {
509  for (int i = 0; i < argc; ++i) {
510  data->col->push_back(std::string(col_name[i] ? col_name[i] : "NULL"));
511  }
512  }
513 
514  // get result
515  SQLITE_ROW_VECTOR row;
516  for (int i = 0; i < argc; i++) {
517  row.push_back(std::string(argv[i] ? argv[i] : "NULL"));
518  }
519 
520  // push SQLITE_ROW_VECTOR to result vector
521  data->rows->push_back(row);
522 
523  return 0;
524  }
